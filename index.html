    function allowDrop(ev) { ev.preventDefault(); }


    function drag(ev) { 

      ev.dataTransfer.setData("text", ev.target.id); 

    }


    // FIXED DROP FUNCTION - This resolves the "Cannot read properties of null (reading 'cloneNode')" error

    function drop(ev) {

      ev.preventDefault();

      const data = ev.dataTransfer.getData("text");

      if (!data) return;


      const original = document.getElementById(data);

      if (!original) return;


      const clone = original.cloneNode(true);

      clone.id = "cloned_" + Date.now();

      clone.draggable = true;

      clone.ondragstart = drag;


      ev.target.innerHTML = "";   // Clear any previous item

      ev.target.appendChild(clone);

    }
